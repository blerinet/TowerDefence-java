6   package fi.towerdefencegamesinc.towerdefence.java.logic;
7   
8   import java.util.Date;
9   
10  /**
11   * A score object with information related to a single score.
12   *
13   * @author vrsaari
14   */
15  public class Score implements Comparable {
16  
17      private final Long score;
18      private final Date date;
19      private final Player player;
20  
21      /**
22       * Create a score object. It uses the current time as a timestamp for it.
23       *
24       * @param name Name of the player performing the score.
25       * @param score The score to be recorded.
26       */
27      public Score(String name, long score) {
28          this.player = new Player(name);
29          this.score = score;
30          this.date = new Date();
31      }
32  
33      @Override
34      public String toString() {
35          return String.format("%1$s10 (%3$te2.%3$tm2.%3$tY4): %2$s10", player.getName(), score, date);
36      }
37  
38      public Date getDate() {
39          return date;
40      }
41  
42      public Player getPlayer() {
43          return player;
44      }
45  
46      public long getScore() {
47          return score;
48      }
49  
50      @Override
51      public int compareTo(Object o) {
52          Score other = (Score) o;
53          return this.score.compareTo(other.score);
54      }
55  
56  }
