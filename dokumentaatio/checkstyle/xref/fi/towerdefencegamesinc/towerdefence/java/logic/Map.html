6   package fi.towerdefencegamesinc.towerdefence.java.logic;

8   import java.io.IOException;
9   import java.nio.file.Files;
10  import java.nio.file.Paths;
11  import java.util.stream.IntStream;
12  import java.util.stream.Stream;

14  /**
15   * The map is used to track objects in game and locations available for
16   * building.
17   *
18   * @author vrsaari
19   */
20  public class Map {
21  
22      Tile[][] tiles;
23      ScoreBoard scoreBoard;
24  
25      /**
26       * Create a new Map object with given width and height.
27       *
28       * @param width The width of the map in tiles.
29       * @param height The height of the map in tiles.
30       */
31      public Map(int width, int height) {
32          this.tiles = new Tile[width][height];
33          this.scoreBoard = new ScoreBoard();
34      }
35  
36      /**
37       * Create a new map object with fixed width and height.
38       *
39       */
40      public Map() {
41          this(10, 10);
42      }
43  
44      /**
45       * Read information from a file to generate a map.
46       *
47       * @param fileName File to read map information from.
48       * @return Generated map object.
49       */
50      public static Map loadMapFromFile(String fileName) {
51          char[][] tmpTiles;
52          try (Stream<String> stream = Files.lines(Paths.get(fileName))) {
53              tmpTiles = stream.map(s -> s.toCharArray()).toArray(char[][]::new);
54          } catch (IOException e) {
55              e.printStackTrace(System.out);
56              return null;
57          }
58          Map map = loadMapFromCharArray(tmpTiles);
59  
60          return map;
61      }
62  
63      /**
64       * Mostly a assisting method for generating a map from a file.
65       *
66       * @param tmpTiles 2d char-array used to generate a map from.
67       * @return Generated map object.
68       */
69      public static Map loadMapFromCharArray(char[][] tmpTiles) {
70          int width = tmpTiles[0].length;
71          int height = tmpTiles.length;
72          Map map = new Map(width, height);
73          IntStream.range(0, height).forEach(i -> {
74              IntStream.range(0, width).forEach(j -> {
75                  switch (tmpTiles[i][j]) {
76                      case ' ':
77                          map.tiles[i][j] = new Tile(Type.Road, false);
78                          break;
79                      case '#':
80                          map.tiles[i][j] = new Tile(Type.Unbuildable, false);
81                          break;
82                      case '@':
83                          map.tiles[i][j] = new Tile(Type.Buildable, true);
84                          break;
85                      case 'X':
86                          map.tiles[i][j] = new Tile(Type.Spawn, false);
87                          break;
88                      default:
89                          break;
90                  }
91  
92              });
93          });
94          return map;
95      }
96  
97      @Override
98      public String toString() {
99          StringBuilder sb = new StringBuilder();
100         Stream.of(tiles).forEach(x -> {
101             Stream.of(x).forEach(tile -> {
102                 Tile t = (Tile) tile;
103                 if (t.getTower() == null) {
104                     if (null != t.getType()) {
105                         switch (t.getType()) {
106                             case Road:
107                                 sb.append(Type.Road.toString());
108                                 break;
109                             case Unbuildable:
110                                 sb.append(Type.Unbuildable.toString());
111                                 break;
112                             case Buildable:
113                                 sb.append(Type.Buildable.toString());
114                                 break;
115                             case Spawn:
116                                 sb.append(Type.Spawn.toString());
117                                 break;
118                             default:
119                                 break;
120                         }
121                     }
122                 } else {
123                     sb.append(t.getTower().getCharRepr());
124                 }
125             });
126             sb.append("\n");
127         });
128         return sb.toString();
129     }

131     /**
132      *
133      * @return The scoreboard for the map.
134      */
135     public ScoreBoard getScoreBoard() {
136         return scoreBoard;
137     }

139     /**
140      *
141      * @param x x-coordinate for a tile on the map.
142      * @param y y-coordinate for a tile on the map.
143      * @return The map tile corresponding to the x,y given.
144      */
145     public Tile getTile(int x, int y) {
146         return this.tiles[x][y];
147     }
148 
149     /**
150      *
151      * @param x x-coordinate for a tile on the map.
152      * @param y y-coordinate for a tile on the map.
153      * @param t The tile object used to replace the tile in the given location.
154      */
155     public void setTile(int x, int y, Tile t) {
156         this.tiles[x][y] = t;
157     }
158 
159     public void setTiles(Tile[][] tiles) {
160         this.tiles = tiles;
161     }
162 
163     public Tile[][] getTiles() {
164         return tiles;
165     }
