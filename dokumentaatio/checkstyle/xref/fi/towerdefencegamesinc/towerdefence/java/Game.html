22  public class Game {
23  
24      private final GameMap map;
25      private final Player player;
26      private int score;
27  
28      public Game(String mapFile, String playerName, int startCurrency) {
29          this.map = GameMap.loadMapFromFile(mapFile);
30          this.player = new Player(playerName, startCurrency);
31          this.score = 0;
32      }
33  
34      public int getScore() {
35          return score;
36      }
37  
38      public void setScore(int score) {
39          this.score = score;
40      }
41  
42      public Player getPlayer() {
43          return player;
44      }
45  
46      public GameMap getMap() {
47          return map;
48      }
49  
50      public void update() {
51          if (player.gameOver()) {
52              System.out.println("------------\n\n\n\n\n\nGAME OVER\n\n\n\n\n\n------------");
53              return;
54          }
55          List<Attacker> attackers = map.getAllAttackers();
56          List<Attacker> attackersInBase = attackers.stream()
57                  .filter(x -> x.getTile().isBase()).collect(Collectors.toList());
58          int damageToBase = attackersInBase.stream()
59                  .mapToInt(Attacker::attack).sum();
60          if (damageToBase > 0) {
61              this.player.takeDamage(damageToBase);
62          }
63          attackers.removeAll(attackersInBase);
64          System.out.println("Attackers: " + attackers.size());
65          attackers.stream().forEach(a -> a.move());
66          List<Tower> towers = map.getAllTowers();
67          System.out.println("Towers: " + towers.size());
68          towers.stream().forEach(t -> {
69              Tower tower = (Tower) t;
70              System.out.println("Checking tower range: " + tower.getCharRepr() + " " + tower.toString());
71              Attacker target = tower.getTarget(attackers);
72              if (target != null) {
73                  System.out.println("shooting at target");
74                  tower.shoot(target);
75              } else {
76                  System.out.println("Target is null");
77              }
78          });
79          if (new Random().nextInt(3) == 0) {
80              this.spawnAttacker();
81          }
82      }
83  
84      public boolean spawnAttacker() {
85          Tile spawn = this.map.getRandomSpawn();
86          System.out.println("Spawnpoint : " + spawn.toString());
87          Attacker attacker = new BasicAttacker(spawn);
88          spawn.addAttacker(attacker);
89          return true;
90      }

92  }
